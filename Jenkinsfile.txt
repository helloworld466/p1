pipeline{
	agent any
	stages{
		stage('Source Code Management'){
			steps{
				echo 'Pulled from GIT'
			}
		}
	}
	stage('Build'){
		environment{
        		scannerHome = tool 'sonarqube'
    		}
	
		steps{
        		withSonarQubeEnv('sonarqube') {
            			sh 'mvn clean package sonar:sonar'
        		}
        		
			timeout(time: 10, unit: 'MINUTES') {
            			waitForQualityGate abortPipeline: true
        		}
    		}
	}
}
